1. Install
Debian/Ubuntu: sudo apt-get install vim
Redhat/CentOS: yum insatll vim
2. Configure
vim ~/.vimrc  / source ~/.vimrc

set nu
set shortmess=atI
set ruler
set laststatus=2
set foldenable
set foldmethod=manual
set langmenu=zh_CN.UTF-8
set helplang=cn
set encoding=utf-8
set confirm
set autoindent
set cindent
set tabstop=4
set softtabstop=4
set shiftwidth=4
set smarttab
set showmatch
set matchtime=1
set showcmd
